    # Import necessary libraries
    import pandas as pd
    from sklearn.cluster import KMeans
    from sklearn.preprocessing import StandardScaler
    import matplotlib.pyplot as plt
    
    # Load dataset from CSV file
    data = pd.read_csv("data.csv")  # Replace 'data.csv' with the path to your CSV file
    
    # Inspect the dataset
    print(data.head())
    
    # Feature selection
    X = data[['feature1', 'feature2', 'feature3']]  # Example features
    
    # Normalize the data (important for K-Means)
    scaler = StandardScaler()
    X_scaled = scaler.fit_transform(X)
    
    # Initialize the K-Means model
    model = KMeans(n_clusters=3, random_state=42)
    
    # Fit the model
    model.fit(X_scaled)
    
    # Get cluster centers and predictions
    centers = model.cluster_centers_
    labels = model.labels_
    
    # Plot the clusters
    plt.scatter(X_scaled[:, 0], X_scaled[:, 1], c=labels, cmap='viridis')
    plt.scatter(centers[:, 0], centers[:, 1], s=300, c='red', marker='X')  # Cluster centers
    plt.title("K-Means Clustering")
    plt.show()
